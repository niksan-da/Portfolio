def share_test(event, x, y):
    alpha = 0.05
    users_in_event = np.array([logs_clean_funnel.query('exp_id == @x').\
                               groupby('event').agg(events=('user_id', 'count'), users=('user_id', 'nunique')).\
                               loc[event, 'users'],
                               logs_clean_funnel.query('exp_id == @y').\
                               groupby('event').agg(events=('user_id', 'count'), users=('user_id', 'nunique')).\
                               loc[event, 'users']])
    users_in_group = np.array([logs_clean_funnel.groupby('exp_id').agg(users=('user_id', 'nunique')).\
                               loc[x, 'users'],\
                               logs_clean_funnel.groupby('exp_id').agg(users=('user_id', 'nunique')).\
                               loc[y, 'users']])
    p1 = users_in_event[0]/users_in_group[0]
    p2 = users_in_event[1]/users_in_group[1]
    p_combined = (users_in_event[0] + users_in_event[1]) / (users_in_group[0] + users_in_group[1])
    difference = p1 - p2
    z_value = difference / mth.sqrt(p_combined * (1 - p_combined) * (1/users_in_group[0] + 1/users_in_group[1]))
    distr = st.norm(0, 1) 
    p_value = (1 - distr.cdf(abs(z_value))) * 2
    if p_value < alpha:
        print('  :     .')
    else:
        print('     ,  , .')
    
    return print(': {0}, : {1}  {2}. p-: {3}'.format(event, x, y, p_value))
